const std = @import("std");

const w4 = @import("wasm4.zig");

const App = struct{
    buffer: [100]u8 = undefined,
    fba: std.heap.FixedBufferAllocator,
    allocator: *std.mem.Allocator,

    frame: u32,

    blinks: u8,

    gamepad_prev_state: u8,
};

var app: App = .{
    .buffer = undefined,
    .fba = undefined,
    .allocator = undefined,

    .frame = 0,

    .blinks = 0,

    .gamepad_prev_state = 0,
};

const smiley = [8]u8{
    0b11000011,
    0b10000001,
    0b00100100,
    0b00100100,
    0b00000000,
    0b00100100,
    0b10011001,
    0b11000011,
};

export fn start() void {
    app.fba = std.heap.FixedBufferAllocator.init(&app.buffer);
    app.allocator = &app.fba.allocator;

    // w4.trace("Hello from Zig!");

    _ = w4.diskr(&app.blinks, 1);
}

export fn update() void {
    w4.DRAW_COLORS.* = 2;
    w4.text("Hello from Zig!", 10, 10);

    w4.line(10, 10, 50, 50);
    w4.oval(10, 40, 12, 25);
    w4.rect(40, 30, 60, 20);
    w4.vline(120, 5, 80);
    w4.hline(60, 25, 80);

    const gamepad = w4.GAMEPAD1.*;

    if (gamepad & w4.BUTTON_LEFT != 0) {
        w4.tone(262, 60, 100, w4.TONE_PULSE1);
    }

    if (w4.MOUSE_BUTTONS.* != 0) {
        w4.text("Mouse button pressed", 0, 160 - 20);
    }

    const mouse_text = std.fmt.allocPrint(
        app.allocator,
        "Mouse: ({d}, {d})",
        .{ w4.MOUSE_X.*, w4.MOUSE_Y.* },
    ) catch unreachable;
    defer app.allocator.free(mouse_text);
    w4.text(mouse_text, 0, 160 - 10);

    if (gamepad & w4.BUTTON_1 != 0) {
        w4.DRAW_COLORS.* = 4;
    }
    w4.blit(&smiley[0], 76, 76, 8, 8, w4.BLIT_1BPP);
    w4.text("Press X to blink", 16, 90);

    const just_pressed: u8 = w4.GAMEPAD1.* & (w4.GAMEPAD1.* ^ app.gamepad_prev_state);
    app.gamepad_prev_state = w4.GAMEPAD1.*;

    if (just_pressed & w4.BUTTON_1 != 0) {
        app.blinks += 1;
        _ = w4.diskw(&app.blinks, 1);
    }

    const blinks_text = std.fmt.allocPrint(
        app.allocator,
        "Blinks persisted: {d}",
        .{ app.blinks },
    ) catch unreachable;
    defer app.allocator.free(blinks_text);
    w4.text(blinks_text, 0, 100);

    const animFrame = (app.frame / 2) % 3;

    w4.DRAW_COLORS.* = 0x2130;
    w4.blitSub(&bunny[0], 76, 56, 16, 16, animFrame*16, 0, bunny_width, w4.BLIT_2BPP);

    app.frame += 1;
}
